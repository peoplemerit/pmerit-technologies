              'aixord-chatgpt-pack', 'aixord-gemini-pack', 'aixord-copilot-pack',
              'aixord-builder-bundle', 'aixord-complete')

foreach ($pkg in $packages) {
    $zip = [System.IO.Compression.ZipFile]::OpenRead("$pkg.zip")

    $zip.Dispose()
    Write-Host "$pkg : $(if($hasDisclaimer){'✅ PASS'}else{'❌ FAIL'})"
}
